# Translations

You can import only the translations that you need

<template>
  <DatePicker :language="en"></DatePicker>
</template>
<script>
import { en } from '@sum.cumo/vue-datepicker/dist/locale/index.esm'

export default {
  data() {
    return {
      en: en,
    }
  },
}
</script>
